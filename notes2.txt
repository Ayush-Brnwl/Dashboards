WITH source_events AS (
  SELECT
    event_time,
    action,
    actor,
    org,
    repository,
    affected_identity,
    affected_identity_user_id
  FROM `db-dev-l4to-mp-github.github_governance.github_audit_events_normalized`
  WHERE event_time >= TIMESTAMP_SUB(CURRENT_TIMESTAMP(), INTERVAL 180 DAY)
),

classified_candidates AS (
  SELECT
    action,

    CASE
      -- Push protection / secret protection disablement
      WHEN REGEXP_CONTAINS(action, r'push_protection')
        AND REGEXP_CONTAINS(action, r'(disable|disabled|remove|removed)')
        THEN 'SECRET_PROTECTION_DISABLED'

      WHEN REGEXP_CONTAINS(action, r'push_protection')
        AND REGEXP_CONTAINS(action, r'(bypass|bypass_list)')
        THEN 'PUSH_PROTECTION_BYPASS_CONTROL_CHANGE'

      -- Advanced Security
      WHEN REGEXP_CONTAINS(action, r'advanced_security')
        AND REGEXP_CONTAINS(action, r'(disable|disabled)')
        THEN 'ADVANCED_SECURITY_DISABLED'

      WHEN REGEXP_CONTAINS(action, r'advanced_security')
        AND REGEXP_CONTAINS(action, r'(policy|entity_policy|update)')
        THEN 'ADVANCED_SECURITY_POLICY_CHANGE'

      -- CodeQL / code scanning
      WHEN REGEXP_CONTAINS(action, r'codeql')
        AND REGEXP_CONTAINS(action, r'(disable|disabled)')
        THEN 'CODEQL_DISABLED'

      WHEN REGEXP_CONTAINS(action, r'codeql')
        AND REGEXP_CONTAINS(action, r'(update|updated)')
        THEN 'CODEQL_CONFIGURATION_CHANGE'

      WHEN REGEXP_CONTAINS(action, r'code_scanning')
        AND REGEXP_CONTAINS(action, r'(delete|deleted|disable|disabled|dismissal)')
        THEN 'CODE_SCANNING_CONTROL_CHANGE'

      -- Security configuration
      WHEN REGEXP_CONTAINS(action, r'security_configuration')
        AND REGEXP_CONTAINS(action, r'(remove|removed|failed|failure)')
        THEN 'SECURITY_CONFIGURATION_REMOVED_OR_FAILED'

      WHEN REGEXP_CONTAINS(action, r'security_configuration')
        AND REGEXP_CONTAINS(action, r'(apply|applied)')
        THEN 'SECURITY_CONFIGURATION_APPLIED'

      -- Dependabot / repository security updates
      WHEN REGEXP_CONTAINS(action, r'(dependabot|security_updates)')
        AND REGEXP_CONTAINS(action, r'(disable|disabled)')
        THEN 'DEPENDABOT_OR_SECURITY_UPDATES_DISABLED'

      -- Identity guardrails
      WHEN action IN (
        'org.disable_two_factor_requirement',
        'org.disable_saml'
      )
        THEN 'IDENTITY_GUARDRAIL_DISABLED'

      -- Repository exposure
      WHEN REGEXP_CONTAINS(action, r'(visibility|access)')
        AND REGEXP_CONTAINS(action, r'(public|internal|change)')
        THEN 'REPOSITORY_EXPOSURE_CHANGE'

      -- Repository access
      WHEN action IN (
        'repo.add_member',
        'repo.add_team',
        'repo.remove_member',
        'repo.remove_team'
      )
        THEN 'REPOSITORY_ACCESS_CHANGE'

      -- Workflow / checks
      WHEN REGEXP_CONTAINS(action, r'(actions|workflow|checks)')
        AND REGEXP_CONTAINS(action, r'(disable|disabled|policy|auto_trigger)')
        THEN 'WORKFLOW_OR_CHECK_CONTROL_CHANGE'

      -- Audit integrity / streaming
      WHEN REGEXP_CONTAINS(action, r'(audit|stream|streaming|retention)')
        AND REGEXP_CONTAINS(action, r'(disable|disabled|delete|deleted|update|updated)')
        THEN 'AUDIT_EVIDENCE_CONTROL_CHANGE'

      ELSE NULL
    END AS proposed_event_group,

    CASE
      WHEN REGEXP_CONTAINS(action, r'push_protection')
        AND REGEXP_CONTAINS(action, r'(disable|disabled|remove|removed)')
        THEN 'CRITICAL'

      WHEN REGEXP_CONTAINS(action, r'advanced_security')
        AND REGEXP_CONTAINS(action, r'(disable|disabled)')
        THEN 'CRITICAL'

      WHEN REGEXP_CONTAINS(action, r'codeql')
        AND REGEXP_CONTAINS(action, r'(disable|disabled)')
        THEN 'CRITICAL'

      WHEN REGEXP_CONTAINS(action, r'security_configuration')
        AND REGEXP_CONTAINS(action, r'(remove|removed)')
        THEN 'CRITICAL'

      WHEN action IN (
        'org.disable_two_factor_requirement',
        'org.disable_saml'
      )
        THEN 'CRITICAL'

      WHEN REGEXP_CONTAINS(action, r'(dependabot|security_updates)')
        AND REGEXP_CONTAINS(action, r'(disable|disabled)')
        THEN 'HIGH'

      WHEN REGEXP_CONTAINS(action, r'push_protection')
        AND REGEXP_CONTAINS(action, r'(bypass|bypass_list)')
        THEN 'HIGH'

      WHEN REGEXP_CONTAINS(action, r'code_scanning')
        AND REGEXP_CONTAINS(action, r'(delete|deleted|disable|disabled|dismissal)')
        THEN 'HIGH'

      WHEN REGEXP_CONTAINS(action, r'security_configuration')
        AND REGEXP_CONTAINS(action, r'(failed|failure)')
        THEN 'HIGH'

      WHEN REGEXP_CONTAINS(action, r'(actions|workflow|checks)')
        AND REGEXP_CONTAINS(action, r'(disable|disabled|policy|auto_trigger)')
        THEN 'HIGH'

      WHEN REGEXP_CONTAINS(action, r'(audit|stream|streaming|retention)')
        AND REGEXP_CONTAINS(action, r'(disable|disabled|delete|deleted|update|updated)')
        THEN 'HIGH'

      WHEN REGEXP_CONTAINS(action, r'security_configuration')
        AND REGEXP_CONTAINS(action, r'(apply|applied)')
        THEN 'LOW'

      ELSE 'MEDIUM'
    END AS proposed_priority,

    actor,
    org,
    repository,
    event_time
  FROM source_events
)

SELECT
  proposed_event_group,
  proposed_priority,
  action,
  COUNT(*) AS events,
  COUNT(DISTINCT actor) AS distinct_actors,
  COUNT(DISTINCT org) AS affected_org_values,
  COUNT(DISTINCT repository) AS affected_repository_values,
  ARRAY_AGG(DISTINCT actor IGNORE NULLS LIMIT 10) AS sample_actors,
  MIN(event_time) AS first_seen,
  MAX(event_time) AS last_seen
FROM classified_candidates
WHERE proposed_event_group IS NOT NULL
GROUP BY
  proposed_event_group,
  proposed_priority,
  action
ORDER BY
  CASE proposed_priority
    WHEN 'CRITICAL' THEN 1
    WHEN 'HIGH' THEN 2
    WHEN 'MEDIUM' THEN 3
    WHEN 'LOW' THEN 4
    ELSE 5
  END,
  events DESC;
