SELECT
  event_time,
  action,
  actor AS initiated_by,
  user AS affected_identity,
  org,
  repository,
  programmatic_access_type,
  source_file,
  event_key
FROM `db-dev-l4to-mp-github.github_governance.github_audit_events_normalized`
WHERE event_time >= TIMESTAMP('2026-01-01 00:00:00+00')
  AND event_time < TIMESTAMP('2026-07-01 00:00:00+00')
  AND (
    action IN ('user.suspend', 'user.unsuspend')
    OR STARTS_WITH(action, 'external_identity.')
  )
ORDER BY event_time DESC
LIMIT 500;
