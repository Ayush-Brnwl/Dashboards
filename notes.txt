SELECT
  user AS identity_login,
  user_id AS identity_user_id,
  COUNT(*) AS suspension_or_restore_events,
  ARRAY_AGG(DISTINCT action ORDER BY action) AS observed_actions,
  MIN(event_time) AS first_seen,
  MAX(event_time) AS last_seen
FROM `db-dev-l4to-mp-github.github_governance.github_audit_events_normalized`
WHERE event_time >= TIMESTAMP_SUB(CURRENT_TIMESTAMP(), INTERVAL 365 DAY)
  AND action IN ('user.suspend', 'user.unsuspend')
  AND user IS NOT NULL
GROUP BY identity_login, identity_user_id
ORDER BY last_seen DESC;
