UPDATE `db-dev-l4to-mp-github.github_governance.github_governance_events` AS target
SET
  review_priority = exception.priority_override,

  review_reason = CASE
    WHEN STRPOS(
      COALESCE(target.review_reason, ''),
      exception.exception_id
    ) > 0
    THEN target.review_reason

    ELSE CONCAT(
      COALESCE(target.review_reason, ''),
      ' Review priority adjusted from ',
      target.review_priority,
      ' to ',
      exception.priority_override,
      ' by approved automation exception ',
      exception.exception_id,
      ': ',
      exception.reason
    )
  END,

  rule_version = 'v1.1.0',
  classified_at = CURRENT_TIMESTAMP()

FROM `db-dev-l4to-mp-github.github_governance.github_governance_review_exceptions` AS exception

WHERE target.event_time >= TIMESTAMP_SUB(CURRENT_TIMESTAMP(), INTERVAL 180 DAY)
  AND exception.is_active = TRUE
  AND exception.priority_override = 'MEDIUM'

  AND LOWER(TRIM(COALESCE(target.actor, ''))) =
      LOWER(TRIM(exception.actor))

  AND target.action = exception.action

  AND (
    exception.governance_event_type IS NULL
    OR exception.governance_event_type = target.governance_event_type
  )

  -- Update only events that are currently High or were not previously
  -- annotated with this exception.
  AND (
    target.review_priority != exception.priority_override
    OR STRPOS(
      COALESCE(target.review_reason, ''),
      exception.exception_id
    ) = 0
  );
