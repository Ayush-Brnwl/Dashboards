SELECT
  COUNT(*) AS total_rows,
  COUNT(DISTINCT governance_event_id) AS distinct_governance_event_ids,
  COUNT(*) - COUNT(DISTINCT governance_event_id) AS duplicate_rows
FROM `db-dev-l4to-mp-github.github_governance.github_governance_events`;

SELECT
  governance_area,
  governance_event_type,
  review_priority,
  rule_id,
  rule_version,
  COUNT(*) AS events
FROM `db-dev-l4to-mp-github.github_governance.github_governance_events`
GROUP BY
  governance_area,
  governance_event_type,
  review_priority,
  rule_id,
  rule_version
ORDER BY
  governance_area,
  governance_event_type,
  review_priority;

SELECT
  g.event_time,
  g.action,
  g.actor AS initiated_by,
  g.affected_identity,
  g.affected_identity_user_id,
  g.governance_area,
  g.governance_event_type,
  g.review_priority,
  g.review_reason,
  c.identity_login AS matched_critical_identity,
  c.dependency
FROM `db-dev-l4to-mp-github.github_governance.github_governance_events` g
JOIN `db-dev-l4to-mp-github.github_governance.github_governance_critical_identities` c
  ON LOWER(g.affected_identity) = LOWER(c.identity_login)
  OR (
    g.affected_identity_user_id IS NOT NULL
    AND g.affected_identity_user_id = c.identity_user_id
  )
WHERE c.is_active = TRUE
  AND c.criticality = 'HIGH'
  AND g.action IN ('user.suspend', 'user.unsuspend')
ORDER BY g.event_time DESC;
