BEGIN TRANSACTION;

-- 1. Insert or update wildcard LOW-priority exceptions.
MERGE
  `db-dev-l4to-mp-github.github_governance.github_governance_review_exceptions` AS target
USING (
  SELECT *
  FROM UNNEST([
    STRUCT(
      'APPROVED_LZ_FACTORY_ALL_ACTIONS' AS exception_id,
      'ghemu-landing-zone-factory[bot]' AS actor,
      '*' AS action,
      CAST(NULL AS STRING) AS governance_event_type,
      'LOW' AS priority_override,
      'Approved Cloud Platform landing-zone automation. Override non-critical governance events to LOW. CRITICAL security-control events remain CRITICAL through classifier precedence.' AS reason,
      'Cloud Platform' AS owner_team,
      TRUE AS is_active
    ),

    STRUCT(
      'APPROVED_PLATFORM_AUTOMATION_ALL_ACTIONS' AS exception_id,
      'cem401-CPTCLDEG-gh-g_dbk' AS actor,
      '*' AS action,
      CAST(NULL AS STRING) AS governance_event_type,
      'LOW' AS priority_override,
      'Approved Cloud Platform app-repo provisioning automation. Override non-critical governance events to LOW. CRITICAL security-control events remain CRITICAL through classifier precedence.' AS reason,
      'Cloud Platform' AS owner_team,
      TRUE AS is_active
    )
  ])
) AS source

ON target.exception_id = source.exception_id

WHEN MATCHED THEN
  UPDATE SET
    target.actor = source.actor,
    target.action = source.action,
    target.governance_event_type = source.governance_event_type,
    target.priority_override = source.priority_override,
    target.reason = source.reason,
    target.owner_team = source.owner_team,
    target.is_active = source.is_active

WHEN NOT MATCHED THEN
  INSERT (
    exception_id,
    actor,
    action,
    governance_event_type,
    priority_override,
    reason,
    owner_team,
    is_active
  )
  VALUES (
    source.exception_id,
    source.actor,
    source.action,
    source.governance_event_type,
    source.priority_override,
    source.reason,
    source.owner_team,
    source.is_active
  );

-- 2. Deactivate the older action-specific MEDIUM exceptions.
UPDATE
  `db-dev-l4to-mp-github.github_governance.github_governance_review_exceptions`
SET
  is_active = FALSE
WHERE
  LOWER(actor) IN (
    'ghemu-landing-zone-factory[bot]',
    'cem401-cptcldeg-gh-g_dbk'
  )
  AND action IN (
    'repo.change_merge_setting',
    'protected_branch.policy_override'
  )
  AND priority_override = 'MEDIUM'
  AND exception_id NOT IN (
    'APPROVED_LZ_FACTORY_ALL_ACTIONS',
    'APPROVED_PLATFORM_AUTOMATION_ALL_ACTIONS'
  );

COMMIT TRANSACTION;
