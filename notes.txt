SELECT
  e.event_time,
  e.action,
  e.actor AS initiated_by,
  e.user AS affected_identity,
  NULLIF(e.user_id, 0) AS affected_identity_user_id,
  c.identity_login AS matched_critical_identity,
  c.identity_user_id AS registered_user_id,
  c.dependency,
  c.owner_team,
  e.source_file
FROM `db-dev-l4to-mp-github.github_governance.github_audit_events_normalized` e
JOIN `db-dev-l4to-mp-github.github_governance.github_governance_critical_identities` c
  ON LOWER(COALESCE(e.user, '')) = LOWER(c.identity_login)
  OR (
    NULLIF(e.user_id, 0) IS NOT NULL
    AND c.identity_user_id IS NOT NULL
    AND NULLIF(e.user_id, 0) = c.identity_user_id
  )
WHERE e.event_time >= TIMESTAMP_SUB(CURRENT_TIMESTAMP(), INTERVAL 180 DAY)
  AND e.action IN ('user.suspend', 'user.unsuspend')
  AND c.is_active = TRUE
  AND c.criticality = 'HIGH'
ORDER BY e.event_time DESC;
