MERGE `db-dev-l4to-mp-github.github_governance.github_governance_review_exceptions` AS target
USING (
  SELECT
    'APPROVED_LZ_FACTORY_BRANCH_POLICY_OVERRIDE' AS exception_id,
    'ghemu-landing-zone-factory[bot]' AS actor,
    'protected_branch.policy_override' AS action,
    'PROTECTED_BRANCH_POLICY_OVERRIDE' AS governance_event_type,
    'MEDIUM' AS priority_override,
    'Approved Cloud Platform landing-zone automation applying standard branch protection during app-repo provisioning.' AS reason,
    'Cloud Platform' AS owner_team,
    TRUE AS is_active

  UNION ALL

  SELECT
    'APPROVED_LZ_FACTORY_MERGE_SETTING_CHANGE',
    'ghemu-landing-zone-factory[bot]',
    'repo.change_merge_setting',
    'MERGE_SETTING_CHANGE',
    'MEDIUM',
    'Approved Cloud Platform landing-zone automation applying standard merge configuration during app-repo provisioning.',
    'Cloud Platform',
    TRUE

  UNION ALL

  SELECT
    'APPROVED_PLATFORM_MERGE_SETTING_CHANGE',
    'cem401-CPTCLDEG-gh-g_dbk',
    'repo.change_merge_setting',
    'MERGE_SETTING_CHANGE',
    'MEDIUM',
    'Approved Cloud Platform automation applying standard merge configuration through the app-repo plugin.',
    'Cloud Platform',
    TRUE

  UNION ALL

  SELECT
    'APPROVED_PLATFORM_BRANCH_POLICY_OVERRIDE',
    'cem401-CPTCLDEG-gh-g_dbk',
    'protected_branch.policy_override',
    'PROTECTED_BRANCH_POLICY_OVERRIDE',
    'MEDIUM',
    'Approved Cloud Platform automation applying standard branch protection through the app-repo plugin.',
    'Cloud Platform',
    TRUE
) AS source
ON target.exception_id = source.exception_id

WHEN MATCHED THEN
  UPDATE SET
    actor                 = source.actor,
    action                = source.action,
    governance_event_type = source.governance_event_type,
    priority_override     = source.priority_override,
    reason                = source.reason,
    owner_team            = source.owner_team,
    is_active             = source.is_active

WHEN NOT MATCHED THEN
  INSERT (
    exception_id,
    actor,
    action,
    governance_event_type,
    priority_override,
    reason,
    owner_team,
    is_active
  )
  VALUES (
    source.exception_id,
    source.actor,
    source.action,
    source.governance_event_type,
    source.priority_override,
    source.reason,
    source.owner_team,
    source.is_active
  );
