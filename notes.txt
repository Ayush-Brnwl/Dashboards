WITH candidate_events AS (
  SELECT
    event_key,
    event_time,
    ingested_at,
    org,
    repository,
    actor,
    action,
    source_file
  FROM `db-dev-l4to-mp-github.github_governance.github_audit_events_normalized`
  WHERE event_time >= TIMESTAMP_SUB(CURRENT_TIMESTAMP(), INTERVAL 7 DAY)
    AND event_time < CURRENT_TIMESTAMP()
    AND event_key IS NOT NULL
    AND action IS NOT NULL
    AND (
      action = 'protected_branch.policy_override'
      OR STARTS_WITH(action, 'protected_branch.')
      OR STARTS_WITH(action, 'required_status_check.')
      OR action = 'repo.change_merge_setting'
      OR action IN (
        'team.add_member',
        'team.remove_member',
        'org.add_member',
        'org.remove_member'
      )
      OR STARTS_WITH(action, 'oauth_access.')
      OR STARTS_WITH(action, 'org_credential_authorization.')
      OR action IN (
        'enterprise.register_self_hosted_runner',
        'org.register_self_hosted_runner'
      )
    )
),

rule_matches AS (
  SELECT
    *,
    CASE
      WHEN action = 'protected_branch.policy_override'
        THEN 'GOV-BRANCH-OVERRIDE'

      WHEN STARTS_WITH(action, 'protected_branch.')
        THEN 'GOV-BRANCH-PROTECTION-CHANGE'

      WHEN STARTS_WITH(action, 'required_status_check.')
        THEN 'GOV-REQUIRED-STATUS-CHECK'

      WHEN action = 'repo.change_merge_setting'
        THEN 'GOV-MERGE-SETTING-CHANGE'

      WHEN action = 'team.add_member'
        THEN 'GOV-TEAM-MEMBER-ADD'

      WHEN action = 'team.remove_member'
        THEN 'GOV-TEAM-MEMBER-REMOVE'

      WHEN action = 'org.add_member'
        THEN 'GOV-ORG-MEMBER-ADD'

      WHEN action = 'org.remove_member'
        THEN 'GOV-ORG-MEMBER-REMOVE'

      WHEN action = 'oauth_access.create'
        THEN 'GOV-OAUTH-ACCESS-CREATE'

      WHEN action = 'oauth_access.regenerate'
        THEN 'GOV-OAUTH-ACCESS-REGENERATE'

      WHEN action = 'oauth_access.destroy'
        THEN 'GOV-OAUTH-ACCESS-DESTROY'

      WHEN STARTS_WITH(action, 'org_credential_authorization.')
        THEN 'GOV-ORG-CREDENTIAL-AUTHORIZATION'

      WHEN action IN (
        'enterprise.register_self_hosted_runner',
        'org.register_self_hosted_runner'
      )
        THEN 'GOV-RUNNER-REGISTRATION'

      ELSE NULL
    END AS rule_id
  FROM candidate_events
),

classified AS (
  SELECT
    *,
    CASE
      WHEN rule_id IN (
        'GOV-BRANCH-OVERRIDE',
        'GOV-BRANCH-PROTECTION-CHANGE',
        'GOV-REQUIRED-STATUS-CHECK',
        'GOV-MERGE-SETTING-CHANGE'
      ) THEN 'BRANCH_AND_MERGE_CONTROLS'

      WHEN rule_id IN (
        'GOV-TEAM-MEMBER-ADD',
        'GOV-TEAM-MEMBER-REMOVE',
        'GOV-ORG-MEMBER-ADD',
        'GOV-ORG-MEMBER-REMOVE'
      ) THEN 'ACCESS_GOVERNANCE'

      WHEN rule_id IN (
        'GOV-OAUTH-ACCESS-CREATE',
        'GOV-OAUTH-ACCESS-REGENERATE',
        'GOV-OAUTH-ACCESS-DESTROY',
        'GOV-ORG-CREDENTIAL-AUTHORIZATION'
      ) THEN 'CREDENTIAL_GOVERNANCE'

      WHEN rule_id = 'GOV-RUNNER-REGISTRATION'
        THEN 'RUNNER_GOVERNANCE'
    END AS governance_area,

    CASE
      WHEN rule_id = 'GOV-BRANCH-OVERRIDE'
        THEN 'PROTECTED_BRANCH_POLICY_OVERRIDE'
      WHEN rule_id = 'GOV-BRANCH-PROTECTION-CHANGE'
        THEN 'BRANCH_PROTECTION_CHANGE'
      WHEN rule_id = 'GOV-REQUIRED-STATUS-CHECK'
        THEN 'REQUIRED_STATUS_CHECK_CHANGE'
      WHEN rule_id = 'GOV-MERGE-SETTING-CHANGE'
        THEN 'MERGE_SETTING_CHANGE'
      WHEN rule_id IN ('GOV-TEAM-MEMBER-ADD', 'GOV-TEAM-MEMBER-REMOVE')
        THEN 'TEAM_MEMBERSHIP_CHANGE'
      WHEN rule_id IN ('GOV-ORG-MEMBER-ADD', 'GOV-ORG-MEMBER-REMOVE')
        THEN 'ORG_MEMBERSHIP_CHANGE'
      WHEN rule_id IN (
        'GOV-OAUTH-ACCESS-CREATE',
        'GOV-OAUTH-ACCESS-REGENERATE',
        'GOV-OAUTH-ACCESS-DESTROY'
      ) THEN 'OAUTH_ACCESS_LIFECYCLE_CHANGE'
      WHEN rule_id = 'GOV-ORG-CREDENTIAL-AUTHORIZATION'
        THEN 'ORG_CREDENTIAL_AUTHORIZATION'
      WHEN rule_id = 'GOV-RUNNER-REGISTRATION'
        THEN 'SELF_HOSTED_RUNNER_REGISTRATION'
    END AS governance_event_type,

    CASE
      WHEN rule_id IN (
        'GOV-BRANCH-OVERRIDE',
        'GOV-MERGE-SETTING-CHANGE'
      ) THEN 'HIGH'

      WHEN rule_id IN (
        'GOV-BRANCH-PROTECTION-CHANGE',
        'GOV-REQUIRED-STATUS-CHECK',
        'GOV-TEAM-MEMBER-ADD',
        'GOV-ORG-MEMBER-ADD',
        'GOV-OAUTH-ACCESS-CREATE',
        'GOV-OAUTH-ACCESS-REGENERATE',
        'GOV-ORG-CREDENTIAL-AUTHORIZATION',
        'GOV-RUNNER-REGISTRATION'
      ) THEN 'MEDIUM'

      ELSE 'LOW'
    END AS review_priority
  FROM rule_matches
  WHERE rule_id IS NOT NULL
)

SELECT
  governance_area,
  governance_event_type,
  review_priority,
  action,
  COUNT(*) AS event_count,
  COUNT(DISTINCT org) AS organizations,
  COUNT(DISTINCT repository) AS repositories,
  COUNT(DISTINCT actor) AS actors
FROM classified
GROUP BY
  governance_area,
  governance_event_type,
  review_priority,
  action
ORDER BY
  governance_area,
  review_priority DESC,
  event_count DESC;
