WITH candidates AS (
  SELECT
    LOWER(TRIM(actor)) AS identity_login,
    NULLIF(actor_id, 0) AS identity_user_id,
    'actor' AS observed_in,
    action,
    event_time
  FROM `db-dev-l4to-mp-github.github_governance.github_audit_events_normalized`
  WHERE event_time >= TIMESTAMP_SUB(CURRENT_TIMESTAMP(), INTERVAL 90 DAY)
    AND REGEXP_CONTAINS(
      LOWER(COALESCE(actor, '')),
      r'(^|[_-])(svc[_-])?ghemu[_-]runner[0-9]*($|[_-])'
    )

  UNION ALL

  SELECT
    LOWER(TRIM(user)) AS identity_login,
    NULLIF(user_id, 0) AS identity_user_id,
    'user' AS observed_in,
    action,
    event_time
  FROM `db-dev-l4to-mp-github.github_governance.github_audit_events_normalized`
  WHERE event_time >= TIMESTAMP_SUB(CURRENT_TIMESTAMP(), INTERVAL 90 DAY)
    AND REGEXP_CONTAINS(
      LOWER(COALESCE(user, '')),
      r'(^|[_-])(svc[_-])?ghemu[_-]runner[0-9]*($|[_-])'
    )
)

SELECT
  identity_login,
  identity_user_id,
  ARRAY_AGG(DISTINCT observed_in ORDER BY observed_in) AS seen_in_fields,
  COUNT(*) AS observed_events,
  ARRAY_AGG(DISTINCT action ORDER BY action LIMIT 15) AS sample_actions,
  MIN(event_time) AS first_seen,
  MAX(event_time) AS last_seen
FROM candidates
GROUP BY identity_login, identity_user_id
ORDER BY last_seen DESC;
