SELECT
  governance_area,
  governance_event_type,
  review_priority,
  COUNT(*) AS events,
  MIN(event_time) AS oldest_event,
  MAX(event_time) AS newest_event
FROM `db-dev-l4to-mp-github.github_governance.github_governance_events`
WHERE event_time >= TIMESTAMP_SUB(CURRENT_TIMESTAMP(), INTERVAL 96 HOUR)
GROUP BY
  governance_area,
  governance_event_type,
  review_priority
ORDER BY
  review_priority,
  events DESC;
