#standardSQL

DECLARE lookback_start TIMESTAMP DEFAULT TIMESTAMP_SUB(@run_time, INTERVAL 96 HOUR);
DECLARE lookback_end TIMESTAMP DEFAULT TIMESTAMP_ADD(@run_time, INTERVAL 5 MINUTE);
DECLARE rule_version STRING DEFAULT 'v1.0.0';

MERGE `db-dev-l4to-mp-github.github_governance.github_governance_events` AS target
USING (
  WITH source_candidates AS (
    SELECT
      event_key,
      event_time,
      ingested_at,
      org,
      repository,
      actor,
      action,
      user AS affected_identity,
      NULLIF(user_id, 0) AS affected_identity_user_id,
      source_file
    FROM `db-dev-l4to-mp-github.github_governance.github_audit_events_normalized`
    WHERE event_time >= lookback_start
      AND event_time < lookback_end
      AND (
        action = 'protected_branch.policy_override'
        OR STARTS_WITH(action, 'protected_branch.')
        OR STARTS_WITH(action, 'required_status_check.')
        OR action IN (
          'repo.change_merge_setting',
          'org.add_member',
          'org.remove_member',
          'team.add_member',
          'team.remove_member',
          'oauth_access.create',
          'oauth_access.regenerate',
          'oauth_access.destroy',
          'user.suspend',
          'user.unsuspend'
        )
      )
    QUALIFY ROW_NUMBER() OVER (
      PARTITION BY event_key
      ORDER BY ingested_at DESC, source_file DESC
    ) = 1
  ),

  active_high_critical_identities AS (
    SELECT
      LOWER(identity_login) AS identity_login,
      identity_user_id,
      dependency
    FROM `db-dev-l4to-mp-github.github_governance.github_governance_critical_identities`
    WHERE is_active = TRUE
      AND criticality = 'HIGH'
    QUALIFY ROW_NUMBER() OVER (
      PARTITION BY LOWER(identity_login)
      ORDER BY
        CASE WHEN identity_user_id IS NULL THEN 1 ELSE 0 END,
        identity_login
    ) = 1
  ),

  enriched_events AS (
    SELECT
      source.*,
      critical.identity_login AS matched_critical_identity,
      critical.dependency AS critical_dependency
    FROM source_candidates AS source
    LEFT JOIN active_high_critical_identities AS critical
      ON LOWER(COALESCE(source.affected_identity, '')) = critical.identity_login
      OR (
        source.affected_identity_user_id IS NOT NULL
        AND critical.identity_user_id IS NOT NULL
        AND source.affected_identity_user_id = critical.identity_user_id
      )
    QUALIFY ROW_NUMBER() OVER (
      PARTITION BY source.event_key
      ORDER BY
        CASE WHEN critical.identity_login IS NULL THEN 1 ELSE 0 END,
        critical.identity_login
    ) = 1
  ),

  mapped_events AS (
    SELECT
      *,

      CASE
        WHEN action = 'protected_branch.policy_override'
          THEN 'BRANCH_AND_MERGE_CONTROLS'

        WHEN STARTS_WITH(action, 'protected_branch.')
          THEN 'BRANCH_AND_MERGE_CONTROLS'

        WHEN STARTS_WITH(action, 'required_status_check.')
          THEN 'BRANCH_AND_MERGE_CONTROLS'

        WHEN action = 'repo.change_merge_setting'
          THEN 'BRANCH_AND_MERGE_CONTROLS'

        WHEN action IN (
          'org.add_member',
          'org.remove_member',
          'team.add_member',
          'team.remove_member',
          'user.suspend',
          'user.unsuspend'
        )
          THEN 'ACCESS_GOVERNANCE'

        WHEN action IN (
          'oauth_access.create',
          'oauth_access.regenerate',
          'oauth_access.destroy'
        )
          THEN 'OAUTH_ACCESS_GOVERNANCE'

        ELSE 'UNCLASSIFIED'
      END AS governance_area,

      CASE
        WHEN action = 'protected_branch.policy_override'
          THEN 'PROTECTED_BRANCH_POLICY_OVERRIDE'

        WHEN STARTS_WITH(action, 'protected_branch.')
          THEN 'BRANCH_PROTECTION_CHANGE'

        WHEN STARTS_WITH(action, 'required_status_check.')
          THEN 'REQUIRED_STATUS_CHECK_CHANGE'

        WHEN action = 'repo.change_merge_setting'
          THEN 'MERGE_SETTING_CHANGE'

        WHEN action IN ('org.add_member', 'org.remove_member')
          THEN 'ORG_MEMBERSHIP_CHANGE'

        WHEN action IN ('team.add_member', 'team.remove_member')
          THEN 'TEAM_MEMBERSHIP_CHANGE'

        WHEN action IN (
          'oauth_access.create',
          'oauth_access.regenerate',
          'oauth_access.destroy'
        )
          THEN 'OAUTH_ACCESS_LIFECYCLE_CHANGE'

        WHEN action = 'user.suspend'
          AND matched_critical_identity IS NOT NULL
          THEN 'CRITICAL_TECHNICAL_ACCOUNT_SUSPENDED'

        WHEN action = 'user.unsuspend'
          AND matched_critical_identity IS NOT NULL
          THEN 'CRITICAL_TECHNICAL_ACCOUNT_RESTORED'

        WHEN action = 'user.suspend'
          THEN 'USER_ACCOUNT_SUSPENDED'

        WHEN action = 'user.unsuspend'
          THEN 'USER_ACCOUNT_RESTORED'

        ELSE 'UNCLASSIFIED'
      END AS governance_event_type,

      CASE
        WHEN action = 'protected_branch.policy_override'
          THEN 'HIGH'

        WHEN action = 'repo.change_merge_setting'
          THEN 'HIGH'

        WHEN action = 'user.suspend'
          AND matched_critical_identity IS NOT NULL
          THEN 'HIGH'

        WHEN action IN (
          'protected_branch.policy_override',
          'repo.change_merge_setting'
        )
          THEN 'HIGH'

        WHEN STARTS_WITH(action, 'protected_branch.')
          THEN 'MEDIUM'

        WHEN STARTS_WITH(action, 'required_status_check.')
          THEN 'MEDIUM'

        WHEN action IN (
          'org.add_member',
          'team.add_member',
          'oauth_access.create',
          'oauth_access.regenerate',
          'user.suspend'
        )
          THEN 'MEDIUM'

        WHEN action IN (
          'org.remove_member',
          'team.remove_member',
          'oauth_access.destroy',
          'user.unsuspend'
        )
          THEN 'LOW'

        ELSE 'LOW'
      END AS review_priority,

      CASE
        WHEN action = 'protected_branch.policy_override'
          THEN 'Protected branch policy was overridden.'

        WHEN STARTS_WITH(action, 'protected_branch.')
          THEN 'Protected branch configuration changed.'

        WHEN STARTS_WITH(action, 'required_status_check.')
          THEN 'Required status-check configuration changed.'

        WHEN action = 'repo.change_merge_setting'
          THEN 'Repository merge configuration changed.'

        WHEN action = 'org.add_member'
          THEN 'Organization membership was granted.'

        WHEN action = 'org.remove_member'
          THEN 'Organization membership was removed.'

        WHEN action = 'team.add_member'
          THEN 'Team membership was granted.'

        WHEN action = 'team.remove_member'
          THEN 'Team membership was removed.'

        WHEN action = 'oauth_access.create'
          THEN 'OAuth access was created.'

        WHEN action = 'oauth_access.regenerate'
          THEN 'OAuth access was regenerated.'

        WHEN action = 'oauth_access.destroy'
          THEN 'OAuth access was revoked.'

        WHEN action = 'user.suspend'
          AND matched_critical_identity IS NOT NULL
          THEN CONCAT(
            'Active critical technical account was suspended. Dependency: ',
            COALESCE(critical_dependency, 'UNSPECIFIED'),
            '.'
          )

        WHEN action = 'user.unsuspend'
          AND matched_critical_identity IS NOT NULL
          THEN CONCAT(
            'Active critical technical account was restored. Dependency: ',
            COALESCE(critical_dependency, 'UNSPECIFIED'),
            '.'
          )

        WHEN action = 'user.suspend'
          THEN 'User account was suspended.'

        WHEN action = 'user.unsuspend'
          THEN 'User account was restored.'

        ELSE 'No v1 governance rule matched.'
      END AS review_reason,

      CASE
        WHEN action = 'protected_branch.policy_override'
          THEN 'BRANCH_POLICY_OVERRIDE'

        WHEN STARTS_WITH(action, 'protected_branch.')
          THEN 'BRANCH_PROTECTION_CHANGE'

        WHEN STARTS_WITH(action, 'required_status_check.')
          THEN 'REQUIRED_STATUS_CHECK_CHANGE'

        WHEN action = 'repo.change_merge_setting'
          THEN 'MERGE_SETTING_CHANGE'

        WHEN action IN ('org.add_member', 'org.remove_member')
          THEN 'ORG_MEMBERSHIP_CHANGE'

        WHEN action IN ('team.add_member', 'team.remove_member')
          THEN 'TEAM_MEMBERSHIP_CHANGE'

        WHEN action IN (
          'oauth_access.create',
          'oauth_access.regenerate',
          'oauth_access.destroy'
        )
          THEN 'OAUTH_ACCESS_LIFECYCLE_CHANGE'

        WHEN action = 'user.suspend'
          AND matched_critical_identity IS NOT NULL
          THEN 'CRITICAL_TECHNICAL_ACCOUNT_SUSPENDED'

        WHEN action = 'user.unsuspend'
          AND matched_critical_identity IS NOT NULL
          THEN 'CRITICAL_TECHNICAL_ACCOUNT_RESTORED'

        WHEN action = 'user.suspend'
          THEN 'USER_ACCOUNT_SUSPENDED'

        WHEN action = 'user.unsuspend'
          THEN 'USER_ACCOUNT_RESTORED'

        ELSE 'UNCLASSIFIED'
      END AS rule_id
    FROM enriched_events
  )

  SELECT
    -- Stable ID by underlying audit event.
    -- This allows later rule improvements to update the same governance record.
    TO_HEX(SHA256(event_key)) AS governance_event_id,

    event_key,
    event_time,
    ingested_at,
    CURRENT_TIMESTAMP() AS classified_at,

    org,
    repository,
    actor,
    affected_identity,
    affected_identity_user_id,

    action,
    governance_area,
    governance_event_type,
    review_priority,
    review_reason,
    rule_id,
    rule_version,
    source_file
  FROM mapped_events
) AS source
ON target.governance_event_id = source.governance_event_id
  AND target.event_time >= lookback_start

WHEN MATCHED AND (
  target.governance_area != source.governance_area
  OR target.governance_event_type != source.governance_event_type
  OR target.review_priority != source.review_priority
  OR IFNULL(target.review_reason, '') != IFNULL(source.review_reason, '')
  OR IFNULL(target.affected_identity, '') != IFNULL(source.affected_identity, '')
  OR IFNULL(
    CAST(target.affected_identity_user_id AS STRING),
    ''
  ) != IFNULL(
    CAST(source.affected_identity_user_id AS STRING),
    ''
  )
  OR target.rule_id != source.rule_id
  OR target.rule_version != source.rule_version
)
THEN UPDATE SET
  classified_at = source.classified_at,
  affected_identity = source.affected_identity,
  affected_identity_user_id = source.affected_identity_user_id,
  governance_area = source.governance_area,
  governance_event_type = source.governance_event_type,
  review_priority = source.review_priority,
  review_reason = source.review_reason,
  rule_id = source.rule_id,
  rule_version = source.rule_version

WHEN NOT MATCHED THEN
  INSERT (
    governance_event_id,
    event_key,
    event_time,
    ingested_at,
    classified_at,
    org,
    repository,
    actor,
    affected_identity,
    affected_identity_user_id,
    action,
    governance_area,
    governance_event_type,
    review_priority,
    review_reason,
    rule_id,
    rule_version,
    source_file
  )
  VALUES (
    source.governance_event_id,
    source.event_key,
    source.event_time,
    source.ingested_at,
    source.classified_at,
    source.org,
    source.repository,
    source.actor,
    source.affected_identity,
    source.affected_identity_user_id,
    source.action,
    source.governance_area,
    source.governance_event_type,
    source.review_priority,
    source.review_reason,
    source.rule_id,
    source.rule_version,
    source.source_file
  );
