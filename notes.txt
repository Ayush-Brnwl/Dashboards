MERGE `db-dev-l4to-mp-github.github_governance.github_governance_critical_identities` AS target
USING (
  WITH service_account_observations AS (
    SELECT
      LOWER(TRIM(actor)) AS identity_login,
      NULLIF(actor_id, 0) AS identity_user_id,
      event_time,
      2 AS source_priority
    FROM `db-dev-l4to-mp-github.github_governance.github_audit_events_normalized`
    WHERE event_time >= TIMESTAMP_SUB(CURRENT_TIMESTAMP(), INTERVAL 180 DAY)
      AND STARTS_WITH(LOWER(TRIM(COALESCE(actor, ''))), 'svc')

    UNION ALL

    SELECT
      LOWER(TRIM(user)) AS identity_login,
      NULLIF(user_id, 0) AS identity_user_id,
      event_time,
      1 AS source_priority
    FROM `db-dev-l4to-mp-github.github_governance.github_audit_events_normalized`
    WHERE event_time >= TIMESTAMP_SUB(CURRENT_TIMESTAMP(), INTERVAL 180 DAY)
      AND STARTS_WITH(LOWER(TRIM(COALESCE(user, ''))), 'svc')
  ),

  canonical_service_accounts AS (
    SELECT
      identity_login,
      identity_user_id
    FROM service_account_observations
    WHERE identity_login IS NOT NULL
      AND identity_login != ''
    QUALIFY ROW_NUMBER() OVER (
      PARTITION BY identity_login
      ORDER BY
        CASE WHEN identity_user_id IS NULL THEN 1 ELSE 0 END,
        source_priority,
        event_time DESC
    ) = 1
  )

  SELECT
    identity_login,
    identity_user_id,
    'TECHNICAL_ACCOUNT' AS identity_type,

    CASE
      WHEN REGEXP_CONTAINS(
        identity_login,
        r'^svc-ghemu-runner[0-9]+_dbk$'
      ) THEN 'HIGH'
      ELSE 'UNASSESSED'
    END AS criticality,

    CASE
      WHEN REGEXP_CONTAINS(
        identity_login,
        r'^svc-ghemu-runner[0-9]+_dbk$'
      ) THEN 'GITHUB_RUNNER_PAT'
      ELSE 'PENDING_VALIDATION'
    END AS dependency,

    CASE
      WHEN REGEXP_CONTAINS(
        identity_login,
        r'^svc-ghemu-runner[0-9]+_dbk$'
      ) THEN 'GitHub Platform'
      ELSE 'UNASSIGNED'
    END AS owner_team,

    TRUE AS is_active
  FROM canonical_service_accounts
) AS source
ON LOWER(COALESCE(target.identity_login, '')) = source.identity_login

WHEN MATCHED THEN
  UPDATE SET
    identity_user_id = COALESCE(source.identity_user_id, target.identity_user_id),
    identity_type    = 'TECHNICAL_ACCOUNT',

    -- Upgrade runner accounts to HIGH, but do not overwrite reviewed classifications
    criticality = CASE
      WHEN source.criticality = 'HIGH' THEN 'HIGH'
      ELSE COALESCE(target.criticality, source.criticality)
    END,

    dependency = CASE
      WHEN source.criticality = 'HIGH' THEN source.dependency
      ELSE COALESCE(target.dependency, source.dependency)
    END,

    owner_team = CASE
      WHEN source.criticality = 'HIGH' THEN source.owner_team
      ELSE COALESCE(target.owner_team, source.owner_team)
    END,

    -- Preserve an explicit manual deactivation
    is_active = COALESCE(target.is_active, source.is_active)

WHEN NOT MATCHED THEN
  INSERT (
    identity_login,
    identity_user_id,
    identity_type,
    criticality,
    dependency,
    owner_team,
    is_active
  )
  VALUES (
    source.identity_login,
    source.identity_user_id,
    source.identity_type,
    source.criticality,
    source.dependency,
    source.owner_team,
    source.is_active
  );

SELECT
  criticality,
  dependency,
  owner_team,
  COUNT(*) AS accounts,
  COUNTIF(identity_user_id IS NOT NULL) AS accounts_with_user_id
FROM `db-dev-l4to-mp-github.github_governance.github_governance_critical_identities`
GROUP BY
  criticality,
  dependency,
  owner_team
ORDER BY
  CASE criticality
    WHEN 'HIGH' THEN 1
    WHEN 'MEDIUM' THEN 2
    WHEN 'LOW' THEN 3
    WHEN 'UNASSESSED' THEN 4
    ELSE 5
  END,
  accounts DESC;
