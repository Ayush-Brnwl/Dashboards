SELECT
  identity_login,
  identity_user_id,
  observed_events,
  sample_actions,
  first_seen,
  last_seen
FROM (
  WITH candidates AS (
    SELECT
      LOWER(TRIM(actor)) AS identity_login,
      NULLIF(actor_id, 0) AS identity_user_id,
      action,
      event_time
    FROM `db-dev-l4to-mp-github.github_governance.github_audit_events_normalized`
    WHERE event_time >= TIMESTAMP_SUB(CURRENT_TIMESTAMP(), INTERVAL 90 DAY)
      AND REGEXP_CONTAINS(
        LOWER(COALESCE(actor, '')),
        r'^svc-ghemu-runner[0-9]+_dbk$'
      )

    UNION ALL

    SELECT
      LOWER(TRIM(user)) AS identity_login,
      NULLIF(user_id, 0) AS identity_user_id,
      action,
      event_time
    FROM `db-dev-l4to-mp-github.github_governance.github_audit_events_normalized`
    WHERE event_time >= TIMESTAMP_SUB(CURRENT_TIMESTAMP(), INTERVAL 90 DAY)
      AND REGEXP_CONTAINS(
        LOWER(COALESCE(user, '')),
        r'^svc-ghemu-runner[0-9]+_dbk$'
      )
  )
  SELECT
    identity_login,
    identity_user_id,
    COUNT(*) AS observed_events,
    ARRAY_AGG(DISTINCT action ORDER BY action LIMIT 20) AS sample_actions,
    MIN(event_time) AS first_seen,
    MAX(event_time) AS last_seen
  FROM candidates
  GROUP BY identity_login, identity_user_id
)
ORDER BY identity_login;



WITH service_account_observations AS (
  SELECT
    LOWER(TRIM(actor)) AS identity_login,
    NULLIF(actor_id, 0) AS identity_user_id,
    'actor' AS seen_in_field,
    action,
    event_time,
    org,
    repository,
    programmatic_access_type,
    source_file
  FROM `db-dev-l4to-mp-github.github_governance.github_audit_events_normalized`
  WHERE event_time >= TIMESTAMP_SUB(CURRENT_TIMESTAMP(), INTERVAL 180 DAY)
    AND STARTS_WITH(LOWER(TRIM(COALESCE(actor, ''))), 'svc')

  UNION ALL

  SELECT
    LOWER(TRIM(user)) AS identity_login,
    NULLIF(user_id, 0) AS identity_user_id,
    'user' AS seen_in_field,
    action,
    event_time,
    org,
    repository,
    programmatic_access_type,
    source_file
  FROM `db-dev-l4to-mp-github.github_governance.github_audit_events_normalized`
  WHERE event_time >= TIMESTAMP_SUB(CURRENT_TIMESTAMP(), INTERVAL 180 DAY)
    AND STARTS_WITH(LOWER(TRIM(COALESCE(user, ''))), 'svc')
),

service_account_summary AS (
  SELECT
    identity_login,
    identity_user_id,
    ARRAY_AGG(DISTINCT seen_in_field ORDER BY seen_in_field) AS seen_in_fields,
    COUNT(*) AS observed_events,
    COUNT(DISTINCT source_file) AS source_files,
    COUNT(DISTINCT org) AS organizations_touched,
    COUNT(DISTINCT repository) AS repositories_touched,

    ARRAY_AGG(
      DISTINCT action
      ORDER BY action
      LIMIT 20
    ) AS sample_actions,

    ARRAY_AGG(
      DISTINCT programmatic_access_type IGNORE NULLS
      ORDER BY programmatic_access_type
      LIMIT 10
    ) AS access_types,

    MIN(event_time) AS first_seen,
    MAX(event_time) AS last_seen
  FROM service_account_observations
  WHERE identity_login IS NOT NULL
    AND identity_login != ''
  GROUP BY identity_login, identity_user_id
)

SELECT
  s.identity_login,
  s.identity_user_id,

  CASE
    WHEN REGEXP_CONTAINS(
      s.identity_login,
      r'^svc-ghemu-runner[0-9]+_dbk$'
    ) THEN 'RUNNER_TECHNICAL_ACCOUNT'
    ELSE 'OTHER_SERVICE_ACCOUNT'
  END AS candidate_type,

  CASE
    WHEN r.identity_login IS NOT NULL THEN 'ALREADY_REGISTERED'
    ELSE 'NOT_REGISTERED'
  END AS registry_status,

  s.seen_in_fields,
  s.observed_events,
  s.source_files,
  s.organizations_touched,
  s.repositories_touched,
  s.access_types,
  s.sample_actions,
  s.first_seen,
  s.last_seen
FROM service_account_summary s
LEFT JOIN `db-dev-l4to-mp-github.github_governance.github_governance_critical_identities` r
  ON LOWER(r.identity_login) = s.identity_login
  OR (
    s.identity_user_id IS NOT NULL
    AND r.identity_user_id = s.identity_user_id
  )
ORDER BY
  registry_status,
  observed_events DESC,
  last_seen DESC;
