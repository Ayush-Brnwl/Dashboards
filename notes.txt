SELECT
  governance_area,
  governance_event_type,
  review_priority,
  rule_version,
  COUNT(*) AS events,
  COUNT(DISTINCT actor) AS actors,
  COUNT(DISTINCT repository) AS repositories,
  MIN(event_time) AS first_seen,
  MAX(event_time) AS last_seen
FROM
  `db-dev-l4to-mp-github.github_governance.github_governance_events`
WHERE
  event_time >= TIMESTAMP_SUB(CURRENT_TIMESTAMP(), INTERVAL 180 DAY)
GROUP BY
  governance_area,
  governance_event_type,
  review_priority,
  rule_version
ORDER BY
  CASE review_priority
    WHEN 'CRITICAL' THEN 1
    WHEN 'HIGH' THEN 2
    WHEN 'MEDIUM' THEN 3
    WHEN 'LOW' THEN 4
    ELSE 5
  END,
  events DESC;


SELECT
  event_time,
  governance_event_type,
  action,
  actor,
  org,
  repository,
  review_priority,
  review_reason
FROM
  `db-dev-l4to-mp-github.github_governance.github_governance_events`
WHERE
  event_time >= TIMESTAMP_SUB(CURRENT_TIMESTAMP(), INTERVAL 180 DAY)
  AND review_priority = 'CRITICAL'
ORDER BY
  event_time DESC
LIMIT 500;


SELECT
  actor,
  review_priority,
  governance_event_type,
  action,
  COUNT(*) AS events
FROM
  `db-dev-l4to-mp-github.github_governance.github_governance_events`
WHERE
  event_time >= TIMESTAMP_SUB(CURRENT_TIMESTAMP(), INTERVAL 180 DAY)
  AND LOWER(actor) IN (
    'ghemu-landing-zone-factory[bot]',
    'cem401-cptcldeg-gh-g_dbk',
    'cem402-cptcldeg-gh-g_dbk'
  )
GROUP BY
  actor,
  review_priority,
  governance_event_type,
  action
ORDER BY
  actor,
  CASE review_priority
    WHEN 'CRITICAL' THEN 1
    WHEN 'HIGH' THEN 2
    WHEN 'MEDIUM' THEN 3
    WHEN 'LOW' THEN 4
    ELSE 5
  END,
  events DESC;
